package main

import "fmt"

// fibonacci is a function that returns
// a function that returns an int.
func fibonacci() func(i int) int {
	f1 := 0
	f2 := 1
	
	return func(i int) int {
		switch i {
		case 0: return 0
		case 1: return 1
		default: 
			sum := f1 + f2
			f1 = f2
			f2 = sum
			return sum
		}
	}
}

func main() {
	f := fibonacci()
	for i := 0; i < 10; i++ {
		fmt.Println(f(i))
	}
}
